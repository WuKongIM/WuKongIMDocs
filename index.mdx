<script
  dangerouslySetInnerHTML={{
    __html: `
      // WuKongIM Language Auto-Detection and Redirection
      (function() {
        'use strict';

        // Configuration
        const CONFIG = {
          STORAGE_KEY: 'wukongim-language-preference',
          SESSION_KEY: 'wukongim-language-selected',
          DEFAULT_LANG: 'en',
          SUPPORTED_LANGS: ['en', 'zh'],
          CHINESE_CODES: ['zh', 'zh-cn', 'zh-tw', 'zh-hk', 'zh-sg', 'zh-mo', 'zh-hans', 'zh-hant']
        };

        // Utility functions
        function getUrlParam(name) {
          const urlParams = new URLSearchParams(window.location.search);
          return urlParams.get(name);
        }

        function isManualSelection() {
          return getUrlParam('manual') === 'true';
        }

        function hasBeenRedirected() {
          try {
            return sessionStorage.getItem(CONFIG.SESSION_KEY) === 'true';
          } catch (e) {
            return false;
          }
        }

        function getSavedPreference() {
          try {
            return localStorage.getItem(CONFIG.STORAGE_KEY);
          } catch (e) {
            return null;
          }
        }

        function setSavedPreference(lang) {
          try {
            localStorage.setItem(CONFIG.STORAGE_KEY, lang);
          } catch (e) {
            // Ignore localStorage errors
          }
        }

        function setRedirectedFlag() {
          try {
            sessionStorage.setItem(CONFIG.SESSION_KEY, 'true');
          } catch (e) {
            // Ignore sessionStorage errors
          }
        }

        function detectBrowserLanguage() {
          // Get browser language with fallbacks
          const browserLang = navigator.language ||
                             navigator.userLanguage ||
                             navigator.browserLanguage ||
                             navigator.systemLanguage ||
                             CONFIG.DEFAULT_LANG;

          const langCode = browserLang.toLowerCase();

          // Check if it's a Chinese variant
          const isChineseVariant = CONFIG.CHINESE_CODES.some(code =>
            langCode === code || langCode.startsWith(code + '-')
          );

          return isChineseVariant ? 'zh' : 'en';
        }

        function redirectToLanguage(lang) {
          // Validate language
          if (!CONFIG.SUPPORTED_LANGS.includes(lang)) {
            lang = CONFIG.DEFAULT_LANG;
          }

          // Disable meta refresh fallback since we're handling redirect
          const metaRefresh = document.getElementById('fallback-redirect');
          if (metaRefresh) {
            metaRefresh.remove();
          }

          // Set flags before redirect
          setSavedPreference(lang);
          setRedirectedFlag();

          // Perform redirect
          const targetUrl = '/' + lang;

          // Use replace to avoid back button issues
          if (window.location.replace) {
            window.location.replace(targetUrl);
          } else {
            window.location.href = targetUrl;
          }
        }

        function shouldRedirect() {
          // Don't redirect if this is a manual selection
          if (isManualSelection()) {
            return false;
          }

          // Don't redirect if already redirected in this session
          if (hasBeenRedirected()) {
            return false;
          }

          // Don't redirect if we're already on a language-specific page
          const currentPath = window.location.pathname;
          if (currentPath.startsWith('/en') || currentPath.startsWith('/zh')) {
            return false;
          }

          return true;
        }

        function init() {
          // Wait for DOM to be ready
          if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', init);
            return;
          }

          // Check if we should redirect
          if (!shouldRedirect()) {
            // Disable meta refresh if we're not redirecting
            const metaRefresh = document.getElementById('fallback-redirect');
            if (metaRefresh) {
              metaRefresh.remove();
            }
            return;
          }

          // Check for saved preference first
          const savedLang = getSavedPreference();
          if (savedLang && CONFIG.SUPPORTED_LANGS.includes(savedLang)) {
            redirectToLanguage(savedLang);
            return;
          }

          // Detect browser language and redirect
          const detectedLang = detectBrowserLanguage();
          redirectToLanguage(detectedLang);
        }

        // Handle manual language selection
        function handleManualSelection() {
          const urlParams = new URLSearchParams(window.location.search);
          const manual = urlParams.get('manual');

          if (manual === 'true') {
            // Clear the manual parameter from URL
            const currentPath = window.location.pathname;
            let targetLang = 'en';

            if (currentPath.startsWith('/zh')) {
              targetLang = 'zh';
            } else if (currentPath.startsWith('/en')) {
              targetLang = 'en';
            }

            // Save the manual selection
            setSavedPreference(targetLang);
            setRedirectedFlag();

            // Clean up URL
            const cleanUrl = currentPath;
            if (window.history && window.history.replaceState) {
              window.history.replaceState({}, document.title, cleanUrl);
            }
          }
        }

        // Immediate initialization (runs before DOM ready for faster redirect)
        function immediateInit() {
          try {
            // Quick check for immediate redirect
            if (shouldRedirect()) {
              const savedLang = getSavedPreference();
              if (savedLang && CONFIG.SUPPORTED_LANGS.includes(savedLang)) {
                redirectToLanguage(savedLang);
                return;
              }

              const detectedLang = detectBrowserLanguage();
              redirectToLanguage(detectedLang);
            }
          } catch (error) {
            // Silently fail, let the main init handle it
          }
        }

        // Initialize immediately
        immediateInit();

        // Also initialize when DOM is ready (fallback)
        try {
          handleManualSelection();
          init();
        } catch (error) {
          console.warn('WuKongIM language detection failed:', error);
          // Fallback: redirect to English if something goes wrong
          if (shouldRedirect()) {
            window.location.href = '/en';
          }
        }
      })();
    `
  }}
/>

# Welcome to WuKongIM

WuKongIM is a high-performance distributed instant messaging service that supports various scenarios including chat applications, message push, IoT communication, audio/video signaling, live streaming, customer service systems, AI communication, and instant communities.

**As simple as Redis, as high-performance as Kafka, as reliable as MySQL**

  dangerouslySetInnerHTML={{
    __html: `
      // Handle manual language selection
      function handleLanguageSelection(lang) {
        try {
          localStorage.setItem('wukongim-language-preference', lang);
          sessionStorage.setItem('wukongim-language-selected', 'true');
        } catch (e) {
          // Ignore storage errors
        }
      }

      // Update UI based on redirect status
      function updateLanguageSelectionUI() {
        const messageElement = document.getElementById('auto-redirect-message');
        const noticeElement = document.getElementById('language-selection-notice');

        if (messageElement && noticeElement) {
          // Check if we're showing this page because auto-redirect failed
          const urlParams = new URLSearchParams(window.location.search);
          const manual = urlParams.get('manual');

          if (manual === 'true') {
            messageElement.textContent = 'Manual language selection';
            noticeElement.style.backgroundColor = '#d1ecf1';
            noticeElement.style.borderColor = '#bee5eb';
          } else {
            // Show a timeout message if still on this page after 3 seconds
            setTimeout(() => {
              if (window.location.pathname === '/' || window.location.pathname === '/index') {
                messageElement.textContent = 'Auto-redirect disabled. Please select manually.';
                noticeElement.style.backgroundColor = '#fff3cd';
                noticeElement.style.borderColor = '#ffeaa7';
              }
            }, 3000);
          }
        }
      }

      // Initialize UI updates
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', updateLanguageSelectionUI);
      } else {
        updateLanguageSelectionUI();
      }
    `
  }}

## Architecture Highlights

- **Unlimited Group Members**: Easily support 100,000+ member group chats with permanent message storage
- **Low Resource Consumption**: Custom binary protocol with 1-byte heartbeat packets for optimal bandwidth and battery usage
- **High Availability**: Modified Raft distributed protocol with automatic disaster recovery and zero-downtime scaling
- **Decentralized**: No single point of failure, every node is independent and equal
